def naive(b, n):
    a = str(b)
    Shares = np.zeros((n, len(a)))
    for i in range(len(a)):
        for j in range(n - 1):
            Shares[j, i] = randint(0, 10)
        Shares[n - 1, i] = (int(a[i]) - sum(Shares[:, i])) % 10
    return Shares


def rec_naive(Shares):
    return (sum(Shares) % 10).astype(int)
